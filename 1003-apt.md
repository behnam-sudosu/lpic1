#### apt  

```bash
#install package
apt

#update list of repository
apt update

#upgrade package
apt upgrade

#install package
apt install openssh-service  

#show firefox in repository
apt-cache search firefox  

#show package must upgrade
apt list --upgradable  

#show library
apt-cache showpkg nginx  

#show dependency
apt-cache depends nginx

#remove dependency
apt autoremove  

#upgrade your os
do-release-upgrade

#change country
vim /etc/apt/sources.list.d
	deb htt://SERVER_IP  

#remove package  
apt remove (name package)  

#uninstall program  
apt purge (name package) 
```
