#### apt  

```bash
#install package
apt

#update list of repository
apt update

#upgrade package
apt upgrade

#install package
apt install openssh-service  

#show firefox in repository
apt-cache search firefox  

#show package must upgrade
apt list --upgradable  

#show library
apt-cache showpkg nginx  

#show dependency
apt-cache depends nginx

#remove dependency
apt autoremove  

#upgrade your os
do-release-upgrade

#change country
vim /etc/apt/sources.list.d
	deb htt://SERVER_IP  

#remove package  
apt remove PACKAGE NAME    

#uninstall program  
apt purge PACKAGE NAME 

#clean cash
apt clean

#remove .deb old
apt autoclean

#directory apt upgrade
/var/cache/apt/archives

#directort apt update
/var/lib/apt/lists
```

## redhat  

#### rpm

```bash
#download
wget LINK OR URL

#for install
rpm -ivh PACKAGE NAME 

#query
rpm -q

#where copy file in directory
rpm -l

#show more information about package
rpm -qip

#where is config file
rpm -qc

#all list
rpm -qa

#delete package
rpm -e

#library needs
rpm -R  

#if installed just upgrade and if isn't installed just install
rpm -U  

#erase
rpm -e
```
